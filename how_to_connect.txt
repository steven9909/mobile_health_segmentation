# --mem - amount of system ram
# --gres - amount of GPUS requesting
# --partition - run info to get a list of partitions - we want to use biggpunodes
# --t - amount of time we can reserve the nodes in minutes (max is 3 days "-t 3-0") 
# -c - number of cpu cores allocated
srun --partition=smallgpunodes -c 1 --mem=2G --gres=gpu:1 -t 1 train.sh

Run sinfo -N -p biggpunodes -o '%10N %G' to see the list of gpunodes
Add in --nodelist gpunode1 to specify which gpu you want to reserve
